/**
Inno3 Pro Example Application
Company: Power Integrations
Summary: This is to demonstrate the Inno3-Pro Arduino Library
Upon Upload initial voltage is set to 5V, then every 1Sec
the voltage will ramp Up by 10mv until it Reaches 15V.
Once 15V was reached, the voltage will ramp down by 10mv every
after 1sec until in goes to 5V. And then Process Repeats
Author: Clomiferds M. Sales
Date: May 18, 2017
*/
//Step 1 : Add the Header Files
#include <Drv_Rtc.h>
#include <Drv_i2c.h>
#include <InnoPro.h>
#include <Config.h>
//Step 2 : Create the class instance
InnoPro_Rtc InnoProClk;
InnoPro_Application InnoProApp;
//Step 3 : Write Initial Commands to Inno Pro
void setup()
{
InnoProApp.InnoPro_Initialization();
}
//Step 4 : Call the Functions on the Main Loop
void loop()
{
// Main Loop Variables
static double fVoltInc = 5; //Initialize Voltage at 5V
static bool bVoltIncrease = true; //Initialize Voltage to Increment
static uint16_t u16_Request_Timer = 0; //Initialize Request Timer
//Timer Routine For Automatic Activation of Requests
if(InnoProClk.clock_HasTimeElapsedMs(u16_Request_Timer,1000)) //Delay Time 1sec
{
InnoProApp.b_Request_Enable = true; //Activate Voltage Change
u16_Request_Timer = InnoProClk.clock_GetTimeStampMs(); //Reset Timer
}
//Increment/Decrement Process
if(InnoProApp.b_Request_Enable)
{
if(bVoltIncrease)//Increase Voltage
{
fVoltInc = fVoltInc + 0.010; //Increment by 10mV
sig_max(fVoltInc,15); //15V Max Saturation
if(fVoltInc == 15) //if 15V reached,
{
bVoltIncrease = false; //Activate Decrement Process
}
}
else //Decrease Voltage
{
fVoltInc = fVoltInc -0.010; //Decrement by 10mV
sig_min(fVoltInc,5); //5V Min Saturation
if(fVoltInc == 5) //if 5V reached,
{
bVoltIncrease = true; //Activate Increment Process
}
}
InnoProApp.b_Request_Enable = false;
}
//Library Call in the Mainloop
InnoProApp.InnoPro_Write_VI(fVoltInc,5); //Set Voltage as variable and Set 3A current
InnoProApp.InnoPro_Write_Cable_Drop_Comp(300);
InnoProApp.InnoPro_Write_Volt_Peak(7);
InnoProApp.InnoPro_Vbus_Switch_Control(1);
}
