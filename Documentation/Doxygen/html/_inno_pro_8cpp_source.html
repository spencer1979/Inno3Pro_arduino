356 bool InnoPro_Application::InnoPro_Process_Voltage(bool bVoltIncrease)
357 {
358  static uint16_t u16_Config_State = 0; //Config State
359  static uint16_t u16_Config_Timer = 0; //Timer for Writing the Configurations
360 
361 
362  if(InnoProTimer.clock_HasTimeElapsedMs(u16_Config_Timer,INNOPRO_VI_PARAM_DELAY)) //Delay Time
363  {
364  if(bVoltIncrease)
365  {
366  switch(u16_Config_State)
367  {
368  //I2C Address , PI COMMAND ADDRESS, //LSB and MSB //I2C COMM TYPE
369  case 0: break; //delay
370  case 1: InnoProSend.I2C_Write16(INNOPRO_ADDRESS , INNOPRO_CC, u8_Buffer_CC, WR_BYTE); break;
371  case 2: InnoProSend.I2C_Write16(INNOPRO_ADDRESS , INNOPRO_OVA, u8_Buffer_OVA, WR_BYTE); break;
372  case 3: InnoProSend.I2C_Write16(INNOPRO_ADDRESS , INNOPRO_CV, u8_Buffer_CV, WR_WORD); break;
373  case 4: InnoProSend.I2C_Write16(INNOPRO_ADDRESS , INNOPRO_UVA, u8_Buffer_UVA, WR_BYTE); break;
374  case 5: InnoProSend.I2C_Write16(INNOPRO_ADDRESS , INNOPRO_UVL, u8_Buffer_UVL, WR_BYTE); break;
375  default: break;
376  }
377  }
378  else
379  {
380  switch(u16_Config_State)
381  {
382  //I2C Address , PI COMMAND ADDRESS, //LSB //MSB //I2C COMM TYPE
383  case 0: break; //delay
384  case 1: InnoProSend.I2C_Write16(INNOPRO_ADDRESS , INNOPRO_CC, u8_Buffer_CC, WR_BYTE); break;
385  case 2: InnoProSend.I2C_Write16(INNOPRO_ADDRESS , INNOPRO_UVL, u8_Buffer_UVL, WR_BYTE); break;
386  case 3: InnoProSend.I2C_Write16(INNOPRO_ADDRESS , INNOPRO_UVA, u8_Buffer_UVA, WR_BYTE); break;
387  case 4: InnoProSend.I2C_Write16(INNOPRO_ADDRESS , INNOPRO_CV, u8_Buffer_CV, WR_WORD); break;
388  case 5: InnoProSend.I2C_Write16(INNOPRO_ADDRESS , INNOPRO_OVA, u8_Buffer_OVA, WR_BYTE); break;
389  default: break;
390  }
391  }
392  u16_Config_State++;
393 
394  u16_Config_Timer = InnoProTimer.clock_GetTimeStampMs(); //Reset Timer
395  }
396 
397  //Check Maximum I2C Write Transactions
398  if(u16_Config_State > INNOPRO_WATCHDOG_DELAY) //Delay
399  {
400  u16_Config_State = 0; //Reset States
401  return true;
402  }
403  else
404  {
405  return false;
406  }
