#include <Drv_Rtc.h>
#include <Drv_i2c.h>
#include <InnoPro.h>
#include <Config.h>
// Variables will change:
static int ledState = LOW; // ledState used to set the LED
InnoPro_Rtc InnoProClk;
InnoPro_Application InnoProApp;
void setup() {
// initialize digital pin LED_BUILTIN as an output.
pinMode(LED_BUILTIN, OUTPUT);
//Serial.begin(9600);
}
bool print_Buffers (void);
void print_Register_values (void);
void loop(){
// Main Loop Variables
static uint16_t u16_Main_State = 0; //Main State Counter
static uint16_t u16_Request_Timer = 0;
if(InnoProClk.clock_HasTimeElapsedMs(u16_Request_Timer,5000)) //Delay Time
{
// if the LED is off turn it on and vice-versa:
ledState ^= 1;
// set the LED with the ledState of the variable:
digitalWrite(LED_BUILTIN, ledState);
u16_Main_State++;
u16_Request_Timer = InnoProClk.clock_GetTimeStampMs();
}
// Main Loop States
switch(u16_Main_State)
{
case 0: u16_Main_State = 1; //Write 5V Default Configuration
InnoProApp.b_Volt_Setting = true; //Initial Voltage Direction
//Write Initial Commands to Inno Pro
InnoProApp.InnoPro_Initialization();
break;
case 1: //Activate 5V Configuration
InnoProApp.InnoPro_Write_VI(5,6);
InnoProApp.InnoPro_Write_Cable_Drop_Comp(300);
InnoProApp.InnoPro_Write_Volt_Peak(7);
InnoProApp.InnoPro_Vbus_Switch_Control(1);
break;
case 2: //Activate 9V Configuration
InnoProApp.InnoPro_Write_VI(9,4);
InnoProApp.InnoPro_Write_Cable_Drop_Comp(200);
InnoProApp.InnoPro_Write_Volt_Peak(7);
InnoProApp.InnoPro_Vbus_Switch_Control(1);
break;
case 3: //Activate 15V Configuration
InnoProApp.InnoPro_Write_VI(15,3);
InnoProApp.InnoPro_Write_Cable_Drop_Comp(100);
InnoProApp.InnoPro_Write_Volt_Peak(9);
InnoProApp.InnoPro_Vbus_Switch_Control(1);
break;
case 4: //Re - Activate 5V Configuration
InnoProApp.InnoPro_Write_VI(5,6);
InnoProApp.InnoPro_Write_Cable_Drop_Comp(300);
InnoProApp.InnoPro_Write_Volt_Peak(7);
InnoProApp.InnoPro_Vbus_Switch_Control(1);
break;
case 5: //Activate 3V Configuration
InnoProApp.InnoPro_Write_VI(3,6);
InnoProApp.InnoPro_Write_Cable_Drop_Comp(300);
InnoProApp.InnoPro_Write_Volt_Peak(7);
InnoProApp.InnoPro_Vbus_Switch_Control(1);
break;
default: // Initialization
u16_Main_State = 1;
break;
}
//print_Buffers();
//print_Register_values();
}
