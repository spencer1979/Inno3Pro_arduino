226 void Inno3Pro_Application::Inno3Pro_Encode_Buffer_Parity( uint16_t u16Temp, uint8_t *u8WriteBuffer)
227 {
228  uint16_t u16TempMsb = 0;
229  uint8_t u8ConvertedMsb = 0;
230  uint8_t u8ConvertedLsb = 0;
231 
232  // Clears Bit 0-6 and Shift the remaining to left by 1
233  // The 7th Bit is used for Parity Purposes
234  // Example for 5V : 01F4 Hex (500 in decimal) , Returns 0x300
235  u16TempMsb = (u16Temp & 0xFF80) << 1;
236 
237  // Begin MSB Extraction
238  // From 0x300 , Returns 0x03
239  u8ConvertedMsb = (u16TempMsb & 0xFF00) >> 8;
240 
241 
242  // Check Odd Parity and Fill the MSB buffer
243  if(Inno3Pro_OddParity(u8ConvertedMsb))
244  {
245  // No of Zero is Odd
246  u8WriteBuffer[1] = u8ConvertedMsb;
247  }
248  else
249  {
250  // No of Zero is Even
251  u8WriteBuffer[1] = set_bit(u8ConvertedMsb,7);
252  }
253 
254 
255  // Clears 7th Bit, This is used for parity purposes
256  // Example for 5V : 01F4 Hex (500 in decimal) , Returns 0x74
257  u8ConvertedLsb = (u16Temp & 0x7F);
258 
259 
260  // Check Odd Parity and Fill the LSB buffer
261  if(Inno3Pro_OddParity(u8ConvertedLsb))
262  {
263  // No of Zero is Odd
264  u8WriteBuffer[0] = u8ConvertedLsb;
265  }
266  else
267  {
268  // No of Zero is Even
269  u8WriteBuffer[0] = set_bit(u8ConvertedLsb,7);
270  }
271 }

433 {

435 
436  if(bEnableVben)

438  fEnableVben = 3; //Vbus ON
439  }

454  Inno3ProSend.I2C_Write16(INNO3PRO_ADDRESS ,INNO3PRO_WATCHDOG_TIMER ,u8_Buffer_WATCHDOG_TIMER ,WR_BYTE);

457  Inno3ProSend.I2C_Write16(INNO3PRO_ADDRESS ,INNO3PRO_VBEN ,u8_Buffer_VBEN ,WR_BYTE);

663 bool Inno3Pro_Application::Inno3Pro_Write_VI(float fSetPtCV, float fSetPtCC)
664 {
665  //Variable for Controlling the Updates
666  static bool bInno3Pro_Allow_Change = false;
667 
668  //Indicates that Update Process was successful
669  static bool bInno3Pro_UpdateDone = false;
670 
671  //Used for Voltage Direction (Increase or Decrease)
672  static bool b_Volt_Setting = true;
673 
674  //Initialize CV
675  static float fPrevious_CV = 0;
676 
677  //Do Not Process Voltage Change Requests Until
678  //the Voltage Transition is Complete
679  if(!bInno3Pro_Allow_Change)
680  {
681  //Check if Voltage Has Changed due to a new CV Request
682  //This Determines the Direction of the Voltage Transition
683  if (fSetPtCV!= fPrevious_CV)
684  {
685  //Compare New CV setting to Previous Setting
686  if(fSetPtCV > fPrevious_CV)
687  {
688  b_Volt_Setting = true; //Increase Voltage
689  }
690  else
691  {
692  b_Volt_Setting = false; //Decrease Voltage
693  }
694 
695  //Save the current CV Value
696  fPrevious_CV = fSetPtCV;
697  }
698 
699  //Start Conversion to Split the Value to LSB and MSB , Implement Parity and Fill up the I2C Buffers
700  Inno3Pro_Encode_Buffer_Parity((uint16_t)Inno3Pro_Compute_CV( fSetPtCV ) ,u8_Buffer_CV);
701  Inno3Pro_Encode_Buffer_Parity((uint16_t)Inno3Pro_Compute_CC( fSetPtCC ) ,u8_Buffer_CC);
702 
703  //Set Over Voltage to 124% of CV
704  Inno3Pro_Encode_Buffer_Parity((uint16_t)Inno3Pro_Compute_OV( fSetPtCV * INNO3PRO_OV_PERCENTAGE_MULT ) ,u8_Buffer_OVA);
705 
706  //Set Under Voltage to 3V
707  Inno3Pro_Encode_Buffer_Parity((uint16_t)Inno3Pro_Compute_UV(INNO3PRO_DEFAULT_UVA_SET_PT_LEVEL) ,u8_Buffer_UVA);
708 
709  //Proceed to New VI Settings, Since Buffers are already Filled
710  bInno3Pro_Allow_Change = true;
711  }
712 
713  //Implement New VI Settings
714  if(bInno3Pro_Allow_Change)
715  {
716  //Auto Disable UVL when less than 3.6V operation
717  if(fSetPtCV <= 3.6 )
718  {
719  //No Response
720  Inno3Pro_Encode_Buffer((uint16_t)INNO3PRO_UVL_FAULT_RESPONSE_NORESPONSE ,u8_Buffer_UVL);
721  }
722  else
723  {
724  //Auto Restart
725  Inno3Pro_Encode_Buffer((uint16_t)INNO3PRO_UVL_FAULT_RESPONSE_AUTORESTART,u8_Buffer_UVL);
726  }
727 
728  //Voltage Transition was Finished?
729  if(Inno3Pro_Process_Voltage(b_Volt_Setting,fSetPtCV) == true)
730  {
731  //Accept New Requests
732  bInno3Pro_Allow_Change = false;
733 
734  //Update process complete
735  bInno3Pro_UpdateDone = true;
736  }
737  }
738 
739  // Return Status
740  if(bInno3Pro_UpdateDone)
741  {
742  return true;
743  }
744  else
745  {
746  return false;
747  }
748 }

752 bool Inno3Pro_Application::Inno3Pro_Process_Voltage(bool bVoltIncrease, float fSetPtCV)
753 {
754  //Control Variables
755  static bool bControlFlag_Increase = false;
756  static bool bVoutpct_Ok_Increase = false;
757 
758  static bool bControlFlag_Decrease= false;
759  static bool bVoutpct_Ok_Decrease = false;
760 
761  //Voltage Increase Routine
762  if(bVoltIncrease)
763  {
764  //Initial Command Sequence
765  if(!bControlFlag_Increase)
766  {
767  //Inno3ProSend.I2C_Write16(INNO3PRO_ADDRESS ,INNO3PRO_UVL ,u8_Buffer_UVL ,WR_BYTE);
768  Inno3ProSend.I2C_Write16(INNO3PRO_ADDRESS ,INNO3PRO_CC ,u8_Buffer_CC ,WR_WORD);
769  Inno3ProSend.I2C_Write16(INNO3PRO_ADDRESS ,INNO3PRO_OVA ,u8_Buffer_OVA ,WR_WORD);
770  Inno3ProSend.I2C_Write16(INNO3PRO_ADDRESS ,INNO3PRO_CV ,u8_Buffer_CV ,WR_WORD);
771 
772  //Debug Pin
773  //digitalWrite(LED_BUILTIN, 0);
774 
775  bControlFlag_Increase = true;
776  }
777 
778  if(bControlFlag_Increase)
779  {
780  //Check If Vout already reached 90% of the desired Set Point
781  if(Inno3Pro_Read_Volts() >(fSetPtCV * 0.9))
782  {
783  //Debug Pin
784  //digitalWrite(LED_BUILTIN, 1);
785 
786  //UVA must be written only after New Voltage Setpoint was reached
787  Inno3ProSend.I2C_Write16(INNO3PRO_ADDRESS ,INNO3PRO_UVA ,u8_Buffer_UVA ,WR_WORD);
788 
789  bVoutpct_Ok_Increase = true;
790  }
791  }
792 
793  //New Set Point Was Reached
794  if(bVoutpct_Ok_Increase)
795  {
796  //Reset all variables
797  bControlFlag_Increase = false;
798  bVoutpct_Ok_Increase = false;
799  return true;
800  }
801  else
802  {
803  return false;
804  }
805  }
806  else //Voltage Decrease Routine
807  {
808  if(!bControlFlag_Decrease)
809  {
810  //Inno3ProSend.I2C_Write16(INNO3PRO_ADDRESS ,INNO3PRO_UVL ,u8_Buffer_UVL ,WR_BYTE);
811  Inno3ProSend.I2C_Write16(INNO3PRO_ADDRESS ,INNO3PRO_CC ,u8_Buffer_CC ,WR_WORD);
812  Inno3ProSend.I2C_Write16(INNO3PRO_ADDRESS ,INNO3PRO_UVA ,u8_Buffer_UVA ,WR_WORD);
813  Inno3ProSend.I2C_Write16(INNO3PRO_ADDRESS ,INNO3PRO_CV ,u8_Buffer_CV ,WR_WORD);
814 
815  // Immediately Executed after CV
816  // Turn on Bleeder
817  Inno3Pro_Bleeder_Enable(true);
818 
819  bControlFlag_Decrease = true;
820 
821  //Debug Pin
822  //digitalWrite(LED_BUILTIN, 0);
823  }
824 
825  //Bleeder Turn Off Control
826  if(bControlFlag_Decrease)
827  {
828  if(!Inno3Pro_Read_Status_Vout2pct())
829  {
830  //Debug Pin
831  //digitalWrite(LED_BUILTIN, 1);
832 
833  //Disable Bleeder
834  Inno3Pro_Bleeder_Enable(false);
835 
836  //OVA must be after Bleeder Turn Off to avoid OVP trigger
837  Inno3ProSend.I2C_Write16(INNO3PRO_ADDRESS ,INNO3PRO_OVA ,u8_Buffer_OVA ,WR_WORD);
838 
839  bVoutpct_Ok_Decrease = true;
840  }
841  }
842 
843  //New Set Point Was Reached
844  if(bVoutpct_Ok_Decrease)
845  {
846  //Reset all variables
847  bControlFlag_Decrease = false;
848  bVoutpct_Ok_Decrease = false;
849  return true;
850  }
851  else
852  {
853  return false;
854  }
855  }
