This the Basic example of how to use the library.

/**
Inno3 Pro Basic Example Application
Company: Power Integrations
Summary: This is to demonstrate the Basic usage of Inno3-Pro Arduino Library
Author: Clomiferds M. Sales
Date: May 18, 2017
*/
//Step 1 : Add the Header Files
#include <Drv_Rtc.h>
#include <Drv_i2c.h>
#include <InnoPro.h>
#include <Config.h>
//Step 2 : Create the class instance
InnoPro_Application InnoProApp;
//Step 3 : Write Initial Commands to Inno Pro
void setup()
{
InnoProApp.InnoPro_Initialization();
}
//Step 4 : Call the Functions on the Main Loop
void loop()
{
// Main Loop Variable Initialization
float fVolts = 5; //Initialize Voltage at 5V
float fAmps = 5; //Initialize Constant Current at 5A
float fCableDropComp = 300; //Initialize Cable Drop Compensation to 300mv
float fVoltPeak = 7; //Initialize Knee Voltage at 7V
float fVbusEn = 1; //Initialize Vbus Enable to at ON
//Library Call in the Mainloop
InnoProApp.InnoPro_Write_VI ( fVolts , fAmps ); //Set Voltage and current
InnoProApp.InnoPro_Write_Cable_Drop_Comp ( fCableDropComp ); //Set Cable Drop Compesation
InnoProApp.InnoPro_Write_Volt_Peak ( fVoltPeak ); //Set Constant Output Power Knee Voltage
InnoProApp.InnoPro_Vbus_Switch_Control ( fVbusEn ); //Set Vbus Enable
}
